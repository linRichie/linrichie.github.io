# 修改网卡名称

  • 主要有两个配置文件

    • 网卡配置文件：/etc/sysconfig/network-scripts/ifcfg-ens160
    • 系统引导文件：/etc/default/grub

  • 修改网卡配置文件

    # 将网卡配置文件重名名
mv /etc/sysconfig/network-scripts/ifcfg-ens160  /etc/sysconfig/network-scripts/ifcfg-eth0

# 修改网卡配置文件里面的内容
vim /etc/sysconfig/network-scripts/ifcfg-eth0 

NAME="eth0"         #将ens33修改为eth0
DEVICE="eth0"       #将ens33修改为eth0

  • 不进入文件进行操作

    sed -i 's/NAME=ens160/NAME=eth0/' /etc/sysconfig/network-scripts/ifcfg-eth0
sed -i 's/DEVICE=ens160/DEVICE=eth0/' /etc/sysconfig/network-scripts/ifcfg-eth0
sed -i 's/BOOTPROTO=dhcp/BOOTPROTO=static/'  /etc/sysconfig/network-scripts/ifcfg-eth0
sed -i 's/ONBOOT=no/ONBOOT=yes/' /etc/sysconfig/network-scripts/ifcfg-eth0

  • 修改系统引导文件

    • 对系统引导文件/etc/default/grub进行修改，在 GRUB_CMDLINE_LINUX 这个参数后面加入 net.ifnames=0 biosdevname=0

    • 不进入文件进行操作

    sed -i 's/quiet/quiet net.ifnames=0 biosdevname=0/' /etc/default/grub

  • 重新生成GRUB配置并更新内核

    grub2-mkconfig -o /boot/grub2/grub.cfg

  • 重启并验证

    reboot
ifconfig

# Kill 命令

# Kill

  • 杀进程

    # kill PID号
kill 20674

  • 强制终止退出

    kill -9 PID
kill -9 20674

  • kill重启进程

kill -HUP PID
ps -aux | grep httpd | awk '{print $2}' | xargs kill -HUP

  • 例子

    ps -ef | grep https | grep -v grep | cut -c 9-15 | xargs kill -s 9

# "grep java"的输出结果是，所有含有关键字"java"的进程。
# "grep -v grep"是在列出的进程中去除含有关键字"grep"的进程。
# "cut -c 9-15"是截取输入行的第9个字符到第15个字符，而这正好是进程号PID。
# "xargs kill -s 9"中的xargs命令是用来把前面命令的输出结果（PID）作为"kill -s 9"命令的参数，并执行该命令。
# "kill  -s 9"会强行杀掉指定进程。

ps -ef | grep httpd | awk '{print $2}' | xargs kill -9

# killall

killall命令杀死同一进程组内的所有进程。其允许指定要终止的进程的名称，而非PID`

killall -9 httpd
killall -9 php-fpm

# pkill

通过程序的名字杀死进程

# pkill踢人用法

# 例子：使用expect实现自动交互

expect << EOF
spawn su root   # spawn表示的是：需要进行的操作
expect "password"  #expect在这里的作用是：屏幕上输出的是否是在这里定义的。用来匹配
send "xxxx\r"   #模拟人键入密码 ； \r 表示的是 回车（确定）
expect eof   #表示的是：执行完后，就关闭这个自动交互
interact #表示的是：执行完成后，保持交互状态。那控制权交给控制台，此时，就可以手动操作
exit
EOF

# curl

# Curl 模拟蜘蛛进行访问

curl -o page1.html http://www.example.com/page/1/ -o page2.html http://www.example.com/page/2/

  • 只显示 HTTP头部信息，不显示网页文件的内容

    curl   --user-agent "Mozilla/5.0 (compatible; Baiduspider/2.0; +http://www.baidu.com/search/spider.html)" -I http://www.example.com/example.html

  • 显示HTTP头部信息和网页文件的内容

    curl   --user-agent "Mozilla/5.0 (compatible; Baiduspider/2.0; +http://www.baidu.com/search/spider.html)" -i http://www.example.com/example/example.html

  • 使用重定向将网页的内容保存到本地文件

    curl   --user-agent "Mozilla/5.0 (compatible; Baiduspider/2.0; +http://www.baidu.com/search/spider.html)" -i http://www.example.com/example/example.html  >index.html

  • 使用-o/-O将文件的内容保存到本地文件

    curl -o index.html  --user-agent "Mozilla/5.0 (compatible; Baiduspider/2.0; +http://www.baidu.com/search/spider.html)" -i http://www.example.com/example/example.html

  • -O URL中的文件名会被用作保存输出的文件名

    curl -O --user-agent "Mozilla/5.0 (compatible; Baiduspider/2.0; +http://www.baidu.com/search/spider.html)" -i http://www.example.com/example/example.html

  • 同时下载多个文件 -o/-O

    curl -O http://www.example.com/page/2/ -O http://www.example.com/page/3/

curl -o page1.html http://www.example.com/page/1/ -o page2.html http://www.example.com/page/2/

# 例子

  • 使用curl请求源站，协议为http，IP为1.1.1.1，域名为www.example.com。

    curl -H "Host:www.example.com" http://1.1.1.1

  • 使用curl请求源站，协议为https，IP为1.1.1.1，域名为www.example.com。并且忽略证书

    curl -k -H "Host:www.example.com" https://1.1.1.1

  • 使用curl请求源站，协议为http，IP为1.1.1.1，域名为www.example.com。只要得到头部信息

    curl -I -H "Host:www.example.com" http://1.1.1.1

  • 使用curl请求源站，协议为http，IP为1.1.1.1，域名为www.example.com。 只要得到头部信息，源站有可能返回了302跳转，要求跟随跳转

    curl -IL -H "Host:www.example.com" http://www.example.com

  • 使用curl请求源站，协议为http，IP为1.1.1.1，域名为www.example.com。 只要得到头部信息，并且显示详细的请求过程

    curl -Ivs -H "Host:www.example.com" http://1.1.1.1
curl --trace output.txt -I -H "Host:www.example.com" http://1.1.1.1

# curl命令判断是否携带SNI

# curl带SNI访问https
curl -v --resolve www.example.com:443:<ipAddress>  https://www.example.com -vv
# 判断curl访问https是否正常 
curl -I -H "Host: www.example.com" https://<ipAddress> -vv

## curl这个命令里面的-v --resolve的具体含义：
## --resolve 后面接的是:  domain:[443]:ipaddress URL(SSL连接：如“https://)、
## --resolve HOST:PORT:ADDRESS  将 HOST:PORT 强制解析到 ADDRESS
## -v, --verbose       显示详细操作信息

# curl 分析源站是否将IP加白

# 案例

# 查找当前目录下的所有文件
find .
# 查找home目录下以.txt结尾的所有文件
find /home -name "*.txt"

find . \( -name "*.txt" -o -name "*.pdf"\)
find . -name  "*.txt" -o -name "*.pdf"

# 匹配文件路径或者文件
# -path参数：指定字符串作为寻找目录的范本样式
find /usr/ -path "*local"

# 基于正则表达式匹配文件路径
find . -regex ".*\(\.txt\|\.pdf\)$"
# 基于正则表达式匹配文件路径,忽略大小写
find . -irehex ".*(\.txt\|\.pdf\)$"

# 否定参数

# 找出home下不是以.txt结尾的文件
find /home ! -name "*.txt"  # ! 参数表示：去反

# 根据文件类型进行搜索

  • find . -type 类型参数
  • 类型参数
    • f 普通文件
    • l 符号连接
    • d 目录
    • c 字符设备
    • b 块设备
    • s 套接字

# 基于目录深度搜索

# 向下最大深度限制为3
find . -maxdepth 3 -type f
# 搜索出深度距离当前目录至少2个子目录的所有文件
find . -mindepth 2 -type f

# 根据文件时间戳进行搜索

  • find . -type f 时间戳
  • UNIX/Linux文件系统每个文件都有三种时间戳：
    • 访问时间（-atime/天，-amin/分钟）：用户最近一次访问时间。
    • 修改时间（-mtime/天，-mmin/分钟）：文件最后一次修改时间。
    • 变化时间（-ctime/天，-cmin/分钟）：文件数据元（例如权限等）最后一次修改时间。
# 搜索最近7天内被访问过的所有文件
find . -type f -atime -7            #(最近7天内)

# 搜索恰好在7天前访问过的所有文件
find . -type f -atime 7             #(恰好在 7 天前)

# 搜索超过7天被访问过的所有文件
find . -type f -atime +7            #(超过7天)

# 搜索访问时间超过10分钟的所有文件
find . -type -f -amin +10        #（超过10分钟）

# 找出比file.log修改时间更长的所有文件
find . -type f -newer file.log      #查找其更改时间较指定文件或目录的更改时间更接近现在的文件或目录

# 根据文件大小进行匹配

find . -type f -size 文件大小单元

  • 文件大小单元:
    • b —— 块（512字节）
    • c —— 字节
    • w —— 字（2字节）
    • k —— 千字节
    • M —— 兆字节
    • G —— 吉字节
# 搜索大于10KB文件
find . -type f -size +10k

# 搜索小于10KB的文件
find . -type f -size -10k

# 搜索等于10KB的文件
find . -type f -size 10K

# 删除匹配文件

# 删除当前目录下的所有.txt文件
fine . -type f -name "*.txt" -delete

# 根据文件权限/所有权进行匹配

# 当前目录下权限为777的文件
find . -type f -perm 777    
#-perm参数：<权限数值>：查找符合指定的权限数值的文件或目录；

# 找出当前目录下权限不是644的php文件
find . -type f -name "*.php" ! -perm 644

#找出当前目录用户tom拥有的文件
find . -type f -user tom 
# user<拥有者名称>：查找符和指定的拥有者名称的文件或目录；

#找出当前目录用户组为sunk拥有的所有文件
find . -type f -group sunk

# 借助-exec选项与其他命令结合使用

# 找出当前目录下所有root的文件，并把所有权更改为用户tom
find . -type f -user root -exec chown tom {} \;
# {}用于与-exec选项结合使用来匹配所有文件，然后会被替换为相应的额文件名

# 找出家目录下所有的.txt文件并删除
find $HOME/. -name "*.txt" -ok rm {} \;
# -ok 和 -exec 行为一样，不会他会给出提示，是否执行相应的操作

# 查找当前目录下所有.txt文件并把他们拼接起来写入到all.txt文件中
find . -type f -name "*.txt" -exec cat {} \; > all.txt

# 将30天前的.log文件移动到old目录中
find . -type f -mtime +30 -name "*.log" -exec cp {} \;

# 将当前目录下所有的.txt文件并以“File:文件名”的形式打印出来
find . -tyep f -name "*.txt" -exec printf "File:%s\n" {} \;

# 由于在单行命令中 -exec参数中无法使用多个命令，下面的方法可以实现在-exec之后接受多条命令，就是在-exec后面执行一个脚本
-exec ./text.sh {} \;

# 搜索但跳出指定的目录

# 查找当前目录或者子目录下所有的.txt文件，但是跳过子目录sk
find . -path "./sk" -prune -o -name "*.txt" -print
# -path <范本样式>：指定字符串作为寻找目录的范本样式；
# -prune：不寻找字符串作为寻找文件或目录的范本样式;

# 列出所有长度为零的文件
find . -empty

# Crontab 定时计划任务

# crontab -e 与/etc/crontab的区别

  • 格式

    • crontab -e的格式：* * * * * command
    • /etc/crontab格式：* * * * * user-name command to be executed

  • 使用范围

    • /etc/crontab：只能root用户可以使用，这种方法更加方便直接给其他用户设置计划任务，而且还可以指定执行shell等。
    • crontba -e：所有用户都可以使用，普通用户也只能为自己设置计划任务，然后自动写入/var/spool/cron/usename

  • 服务重启

/etc/init.d/crond restart
service crond restart
systemctl restart crond
  • crontab会进行语法检查，vi文件不会
  • crontab -e是某个用户的周期计划任务；/etc/crontab是系统的周期任务
  • 不管用 crontab -e or 改 /etc/crontab 都不需要重新启动 crond 服务
01 22 * * * /sbin/shutdown -h now

vi /etc/crontab
* * * * * root /sbin/a.sh -h now
# 或者
* * * * * root  /sbin/a.sh 0

# crontab -e

默认情况下，任何使用只要不被列入到/etc/cron.deny当中，就可以直接使用crontab -e编辑自己的计划任务。
语法：

crontab [-u username] [-l|-e|-r]
# 选项与参数：
# -u  ：只有 root 才能进行这个任务，亦即帮其他使用者创建/移除 crontab 工作排程；
# -e  ：编辑 crontab 的工作内容
# -l  ：查阅 crontab 的工作内容
# -r  ：移除所有的 crontab 的工作内容，若仅要移除一项，请用 -e 去编辑

# 查询使用者目前的crontab内容
crontab -l

# 清空使用者目前的crontab：
crontab -r
crontab -l

# crontab的限制

  • /etc/cron.allow：将可以使用crontab的账号写入其中，若不再这个文件内的使用者则不可使用crontab
  • /etc/cron.deny：将不可以使用crontab的账号写入其中，没有记录到这个文件当中的使用者，就可以使用crontab
  • 优先顺序：/etc/cron.allow比/etc/cron.deny优先。
  • 建议只保存一个文件即可。系统默认是保留/etc/cron.deny/

# /etc/crontab

  • crontab -e是正对使用者设计的。当时系统的例行性任务时，就需要编辑/etc/crontab这个文件。
    /etc/crontab是一个纯文字档，必须用root身份编辑。

  • 组成

    • 前五个*，分别表示：分，时，日，月，周；与crontab -e 是一样的
    • 之后接的是运行命令的身份
    • 最后就是需要执行的command

  • run-parts搜寻看看，其实那是一个 bash script 啦！如果你直接进入 /usr/bin/run-parts 去看看， 会发现这支命令会将后面接的『目录』内的所有文件捉出来运行！这也就是说『 如果你想让系统每小时主动帮你运行某个命令，将该命令写成 script，并将该文件放置到 /etc/cron.hourly/ 目录下即可』的意思！
    现在你知道系统是如何进行他默认的一堆例行性工作排程了吗？如果你下达ll /etc/cron.daily就可以看到一堆文件， 那些文件就是系统提供的 script ，而这堆 scripts 将会在每天的凌晨 6:25 开始运行！

# crontab格式说明

# 格式

分钟：0-59
小时：1-23
日期：1-31
月份：1-12
星期：0-6（0表示周日）

*： 表示任何时刻
,： 表示分割
－：表示一个段，如第二端里： 1-5，就表示1到5点
/n : 表示每个n的单位执行一次，如第二段里，*/1, 就表示每隔1个小时执行一次命令。也可以写成1-23/1.

# & 后台执行命令

  • 在每行最后加上& 则表示从后台执行
  • 但在后台运行是需要注意：
    • 需要用户交互的命令不要放在后台执行，这样会导致机器始终在进行等待
    • 即使在后台运行，一样会把结果输出到屏幕上，所以需要使用使用下面的方式
command > out.file 2>&1 &
2>&1:表示的是：所有的标准输出和错误输出都将被冲顶想到一个叫做out.file的文件中

# 2>&1含义

crontab -l
* * * * * /root/a.sh > /dev/null 2>&1 &
# 就是在后台执行这个命令，并将错误输出2重定向到标准输出1，然后将标准输出1全部放到/dev/null文件中，这个文件也就是清空（黑洞）
# 0表示：键盘输入
# 1表示：标准输出
# 2表示：错误输出
# &1表示：文件描述1，表示标准输出，如果这里少了&就成了数字1，就表示重定向到文件1。

# 2>&1写在后面的原因

  • 格式：command > file 2>&1 == command 1> file 2>&1

  • 首先是command > file将标准输出重定向到file中， 2>&1 是标准错误拷贝了标准输出，也就是同样被重定向到file中，最终结果就是标准输出和错误都被重定向到file中。

  • 如果改成： command 2>&1 >file

  • 2>&1 标准错误拷贝了标准输出的行为，但此时标准输出还是在终端。>file 后输出才被重定向到file，但标准错误仍然保持在终端。
